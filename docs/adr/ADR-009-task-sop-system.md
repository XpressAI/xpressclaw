# ADR-009: Task and SOP System

## Status
Accepted

## Context

Agents need structure for their work:
- **Tasks**: Discrete units of work (like tickets on a kanban board)
- **SOPs (Standard Operating Procedures)**: Reusable workflows that agents can follow

Xpress AI's production system includes a kanban board for tasks. We need to bring this to XpressAI while keeping it simple:
- Agents can create, update, and complete tasks
- Users can view and manage tasks
- SOPs can be created by users or generated by agents
- SOPs can trigger automatically based on conditions

## Decision

We will implement a **task board** with **SOP support**.

### Task Model

```python
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Optional

class TaskStatus(str, Enum):
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    BLOCKED = "blocked"
    COMPLETED = "completed"
    CANCELLED = "cancelled"

class TaskPriority(int, Enum):
    LOW = 0
    NORMAL = 1
    HIGH = 2
    URGENT = 3

@dataclass
class Task:
    id: str
    title: str
    description: Optional[str] = None
    status: TaskStatus = TaskStatus.PENDING
    priority: TaskPriority = TaskPriority.NORMAL
    
    # Ownership
    agent_id: Optional[str] = None
    user_id: Optional[str] = None
    
    # Hierarchy
    parent_task_id: Optional[str] = None
    subtasks: list[str] = field(default_factory=list)
    
    # SOP association
    sop_id: Optional[str] = None
    sop_step: Optional[int] = None
    
    # Timestamps
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    started_at: Optional[datetime] = None
    completed_at: Optional[datetime] = None
    due_at: Optional[datetime] = None
    
    # Metadata
    tags: list[str] = field(default_factory=list)
    context: dict = field(default_factory=dict)  # Arbitrary data
```

### Task Board

```python
class TaskBoard:
    """Manages tasks for all agents."""
    
    def __init__(self, db: Database):
        self.db = db
    
    async def create_task(
        self,
        title: str,
        description: str | None = None,
        agent_id: str | None = None,
        parent_task_id: str | None = None,
        sop_id: str | None = None,
        priority: TaskPriority = TaskPriority.NORMAL,
    ) -> Task:
        """Create a new task."""
        task = Task(
            id=generate_id(),
            title=title,
            description=description,
            agent_id=agent_id,
            parent_task_id=parent_task_id,
            sop_id=sop_id,
            priority=priority,
        )
        
        await self._save_task(task)
        await self._emit_event("task.created", task)
        
        return task
    
    async def update_status(
        self, 
        task_id: str, 
        status: TaskStatus,
        agent_id: str | None = None
    ) -> Task:
        """Update task status."""
        task = await self.get_task(task_id)
        
        # Validate transition
        if not self._is_valid_transition(task.status, status):
            raise InvalidTaskTransition(task.status, status)
        
        task.status = status
        task.updated_at = datetime.now()
        
        if status == TaskStatus.IN_PROGRESS and not task.started_at:
            task.started_at = datetime.now()
            task.agent_id = agent_id
        
        if status == TaskStatus.COMPLETED:
            task.completed_at = datetime.now()
        
        await self._save_task(task)
        await self._emit_event(f"task.{status.value}", task)
        
        return task
    
    async def get_tasks(
        self,
        status: TaskStatus | None = None,
        agent_id: str | None = None,
        limit: int = 100
    ) -> list[Task]:
        """Query tasks with filters."""
        query = "SELECT * FROM tasks WHERE 1=1"
        params = []
        
        if status:
            query += " AND status = ?"
            params.append(status.value)
        
        if agent_id:
            query += " AND agent_id = ?"
            params.append(agent_id)
        
        query += " ORDER BY priority DESC, created_at ASC LIMIT ?"
        params.append(limit)
        
        rows = await self.db.fetch_all(query, params)
        return [self._row_to_task(row) for row in rows]
    
    async def get_next_task(self, agent_id: str) -> Task | None:
        """Get the next task for an agent to work on."""
        # First: tasks already assigned to this agent
        tasks = await self.get_tasks(
            status=TaskStatus.IN_PROGRESS,
            agent_id=agent_id,
            limit=1
        )
        if tasks:
            return tasks[0]
        
        # Second: unassigned pending tasks
        tasks = await self.get_tasks(
            status=TaskStatus.PENDING,
            agent_id=None,
            limit=1
        )
        if tasks:
            # Claim the task
            return await self.update_status(
                tasks[0].id, 
                TaskStatus.IN_PROGRESS,
                agent_id=agent_id
            )
        
        return None
```

### SOP (Standard Operating Procedure)

```python
@dataclass
class SOPStep:
    description: str
    action: str  # What the agent should do
    expected_outcome: Optional[str] = None
    on_failure: Optional[str] = None  # "retry", "skip", "abort", "ask"

@dataclass
class SOPTrigger:
    type: str  # "schedule", "event", "command", "condition"
    value: str  # cron expression, event name, command text, condition
    
@dataclass 
class SOP:
    id: str
    name: str
    description: str
    
    steps: list[SOPStep]
    triggers: list[SOPTrigger] = field(default_factory=list)
    
    # Configuration
    agent_id: Optional[str] = None  # If bound to specific agent
    timeout_minutes: int = 60
    require_confirmation: bool = False
    
    # Metadata
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    created_by: str = "user"  # "user" or agent_id
    
    # Stats
    run_count: int = 0
    success_count: int = 0
    last_run_at: Optional[datetime] = None
```

### SOP Definition Format

SOPs can be defined in YAML:

```yaml
# sops/weekly-report.yaml
name: Weekly Report
description: Compile my weekly summary for the team

triggers:
  - type: schedule
    value: "0 16 * * 5"  # Friday 4pm
  - type: command
    value: "do weekly report"

steps:
  - description: Gather commits and PRs
    action: |
      Review all commits and PRs from the past week.
      Focus on my contributions and team activity.
    expected_outcome: List of commits and PRs with summaries
    
  - description: Check completed tasks
    action: |
      Look at the task board for completed items this week.
    expected_outcome: List of completed tasks
    
  - description: Identify blockers
    action: |
      Note any blockers or concerns that should be raised.
    on_failure: skip  # Not critical if no blockers
    
  - description: Draft email
    action: |
      Compose a concise weekly update email covering:
      - Key accomplishments
      - Completed tasks
      - Any blockers
    expected_outcome: Draft email ready for review
    
  - description: Request review
    action: |
      Ask the user to review the email before sending.
    require_confirmation: true

timeout_minutes: 30
```

### SOP Manager

```python
class SOPManager:
    """Manages SOPs and their execution."""
    
    def __init__(self, db: Database, task_board: TaskBoard):
        self.db = db
        self.task_board = task_board
        self.scheduler = SOPScheduler()
    
    async def load_sop(self, path: Path) -> SOP:
        """Load an SOP from a YAML file."""
        with open(path) as f:
            data = yaml.safe_load(f)
        
        sop = SOP(
            id=generate_id(),
            name=data["name"],
            description=data.get("description", ""),
            steps=[
                SOPStep(
                    description=s["description"],
                    action=s["action"],
                    expected_outcome=s.get("expected_outcome"),
                    on_failure=s.get("on_failure", "abort"),
                )
                for s in data["steps"]
            ],
            triggers=[
                SOPTrigger(type=t["type"], value=t["value"])
                for t in data.get("triggers", [])
            ],
            timeout_minutes=data.get("timeout_minutes", 60),
            require_confirmation=data.get("require_confirmation", False),
        )
        
        await self._save_sop(sop)
        await self._register_triggers(sop)
        
        return sop
    
    async def execute_sop(
        self, 
        sop_id: str, 
        agent_id: str,
        context: dict | None = None
    ) -> SOPExecution:
        """Execute an SOP, creating tasks for each step."""
        sop = await self.get_sop(sop_id)
        
        # Create parent task for the SOP execution
        parent_task = await self.task_board.create_task(
            title=f"SOP: {sop.name}",
            description=sop.description,
            agent_id=agent_id,
            sop_id=sop_id,
        )
        
        # Create subtasks for each step
        for i, step in enumerate(sop.steps):
            await self.task_board.create_task(
                title=f"Step {i+1}: {step.description}",
                description=step.action,
                agent_id=agent_id,
                parent_task_id=parent_task.id,
                sop_id=sop_id,
                context={
                    "sop_step": i,
                    "expected_outcome": step.expected_outcome,
                    "on_failure": step.on_failure,
                }
            )
        
        return SOPExecution(
            sop_id=sop_id,
            parent_task_id=parent_task.id,
            started_at=datetime.now(),
        )
    
    async def create_sop_from_conversation(
        self,
        agent_id: str,
        prompt: str,
        context: list[Message]
    ) -> SOP:
        """Have an agent create an SOP based on conversation."""
```

### SOP Scheduler

```python
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from apscheduler.triggers.cron import CronTrigger

class SOPScheduler:
    """Schedules SOPs based on their triggers."""
    
    def __init__(self):
        self.scheduler = AsyncIOScheduler()
        self.scheduler.start()
    
    async def register_trigger(
        self, 
        sop: SOP, 
        trigger: SOPTrigger,
        callback: Callable
    ) -> None:
        if trigger.type == "schedule":
            # Cron-style schedule
            self.scheduler.add_job(
                callback,
                CronTrigger.from_crontab(trigger.value),
                id=f"sop-{sop.id}-{trigger.value}",
                kwargs={"sop_id": sop.id}
            )
        
        elif trigger.type == "event":
            # Register event listener
            await self.event_bus.subscribe(
                trigger.value,
                lambda e: callback(sop_id=sop.id, event=e)
            )
        
        elif trigger.type == "command":
            # Register command pattern
            await self.command_registry.register(
                trigger.value,
                lambda cmd: callback(sop_id=sop.id, command=cmd)
            )
```

### Agent Integration

Agents interact with tasks through MCP tools:

```python
# tools/builtin/tasks.py

@tool("list_tasks", "List tasks on the board", {"status": str, "limit": int})
async def list_tasks(args: dict, context: AgentContext) -> dict:
    tasks = await context.task_board.get_tasks(
        status=args.get("status"),
        agent_id=context.agent_id,
        limit=args.get("limit", 10)
    )
    return {
        "content": [{"type": "text", "text": format_tasks(tasks)}]
    }

@tool("create_task", "Create a new task", {"title": str, "description": str})
async def create_task(args: dict, context: AgentContext) -> dict:
    task = await context.task_board.create_task(
        title=args["title"],
        description=args.get("description"),
        agent_id=context.agent_id
    )
    return {
        "content": [{"type": "text", "text": f"Created task: {task.id}"}]
    }

@tool("complete_task", "Mark a task as completed", {"task_id": str})
async def complete_task(args: dict, context: AgentContext) -> dict:
    task = await context.task_board.update_status(
        args["task_id"],
        TaskStatus.COMPLETED,
        agent_id=context.agent_id
    )
    return {
        "content": [{"type": "text", "text": f"Completed: {task.title}"}]
    }

@tool("run_sop", "Execute a standard operating procedure", {"sop_name": str})
async def run_sop(args: dict, context: AgentContext) -> dict:
    sop = await context.sop_manager.get_sop_by_name(args["sop_name"])
    execution = await context.sop_manager.execute_sop(
        sop.id,
        context.agent_id
    )
    return {
        "content": [{"type": "text", "text": f"Started SOP: {sop.name}"}]
    }
```

## Consequences

### Positive
- Structured work management for agents
- Reusable procedures via SOPs
- Visibility into agent work
- Supports both user-created and agent-generated SOPs
- Natural integration with memory (context carries over)

### Negative
- Adds complexity to the runtime
- Task/SOP management is another thing to learn
- Scheduling adds external dependency (APScheduler)
